SELECT FNAME, LNAME
FROM EMPLOYEE
WHERE ( (SELECT PNO
 FROM WORKS_ON
 WHERE SSN=ESSN)
 CONTAINS
 (SELECT PNUMBER
 FROM PROJECT
 WHERE DNUM=4) );
ORDER BY Fname
